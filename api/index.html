def extract_from_excels_folder(path: list[str]) -> list[pd.DataFrame]:
    """
    função para ler os arquivos de uma pasta em path e retornar uma lista de DataFrames

    args: input_path (str): caminho da pasta para extração

    return: df_list: lista de DataFrames
    """

    full_path = os.path.join(*path, "*.xlsx")
    path_files = glob.glob(full_path)

    dataframes = [pd.read_excel(file) for file in path_files]
    return dataframes

def concat_dataframes(dataframes: list[pd.DataFrame]) -> pd.DataFrame:
    """
    Função que transforma uma lista de DataFrames em um DataFrame unido.
    """
    concated_df = pd.concat(dataframes, ignore_index=True)
    return concated_df

def load_to_excel(df: pd.DataFrame, filename: str, path: list[str]):
    """
    Salva um DataFrame no formato Excel (xlsx)

    args:
    df: pd.DataFrame: Dataframe que será salvo em Excel
    filename: str: nome do arquivo que será gerado.
    path: list[str]: caminho que será usado para salvar o arquivo.
    """

    folder_path = os.path.join(*path)

    if not os.path.exists(folder_path):
        os.makedirs(folder_path)

    full_path = os.path.join(folder_path, f"{filename}.xlsx")
    df.to_excel(full_path, index=False)
